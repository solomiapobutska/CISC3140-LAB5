app.js 
const express = require('express')
const path = require('path')
const {v4} = require('uuid')
const app = express()

let CONTACTS = [
  {id: v4(), name: 'Solomiya', value: '7188646706', marked: false}
]

app.use(express.json())// allows to work with requests

// GET
app.get('/api/contacts', (req, res) => {
  setTimeout(() => {
    res.status(200).json(CONTACTS)//SEVER is working so transfer contacts
  }, 1000)
})

// POST
app.post('/api/contacts', (req, res) => {
  const contact = {...req.body, id: v4(), marked: false} //based on theedata from client creates new element
  CONTACTS.push(contact)
  res.status(201).json(contact)// new element (201 means element was created)
})

// DELETE
app.delete('/api/contacts/:id', (req, res) => {
  CONTACTS = CONTACTS.filter(c => c.id !== req.params.id) // id to know what element to delete
  res.status(200).json({message: 'Contact was deleted'})
})

// PUT
app.put('/api/contacts/:id', (req, res) => {
  const idx = CONTACTS.findIndex(c => c.id === req.params.id)
  CONTACTS[idx] = req.body// changes a contact in database
  res.json(CONTACTS[idx])
})


app.use(express.static(path.resolve(__dirname, 'client'))) //It is static because otherwise Client won't be able to find frontend'js file
app.get('*', (req, res) => {
  res.sendFile(path.resolve(__dirname, 'client', 'index.html'))
})

app.listen(3000, () => console.log('Server has been started on port 3000...'))
